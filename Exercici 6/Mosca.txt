//canvis (mida)per que s'assimili a la mosca
class Mosca {
  PVector location;
  PVector velocity;
  PVector acceleration;
  float maxSpeed;
  int mida;
  //Creem aquest contructor per defecte
  Mosca() {
    location = new PVector(random(width),random(height));
    velocity = new PVector(random(-2,2),random(-2,2));
    acceleration = new PVector(0.05,0.05);
    maxSpeed = 5; 
    mida = 10;
  }
   
  void update() {
    // Creem els PVectors que contindran l'ubicaci贸 del mouse 
    // i la direcci贸.
    PVector mouse = new PVector(mouseX,mouseY);
    PVector direccio = PVector.sub(mouse,location);
    // Normalitzem la direcci贸.
    direccio.normalize();
    direccio.mult(0.5);
    //Apliquem l'acceleraci贸 i movem l'objecte.
    acceleration = direccio;
    velocity.add(acceleration);
    velocity.limit(maxSpeed);
    location.add(velocity);
  }
  
  void display() {
    stroke(0);
    fill(0,255,0);
    ellipse(location.x,location.y,mida,mida);
  }
  
  void checkEdges() {
    if(location.x > width) {
      location.x = 0;
    }
    else if(location.x < 0) {
      location.x = width;
    }
    
    if(location.y > height) {
      location.y = 0;
    }
    else if(location.y < 0) {
      location.y = height;
    }
  }
  
 
}
